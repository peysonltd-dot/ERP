const express = require('express'); const cors = require('cors'); const { GoogleGenerativeAI } = require('@google/generative-ai'); const app = express(); const port = process.env.PORT || 3000; // 允許所有來源連線 app.use(cors()); // 增加傳輸容量上限，因為要傳圖片 app.use(express.json({ limit: '50mb' })); // 初始化 API (從環境變數讀取鑰匙) const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY); app.post('/api/generate', async (req, res) => { try { const { prompt, imageBase64, mimeType } = req.body; if (!process.env.GEMINI_API_KEY) { return res.status(500).json({ error: '伺服器未設定 API Key' }); } // ⚠️ 修正點：將模型名稱改回最標準的 "gemini-1.5-flash"，拿掉 -latest 就不會 404 了 const model = genAI.getGenerativeModel({ model: "gemini-1.5-flash" }); const parts = [{ text: prompt }]; // 如果有圖片，加進去 if (imageBase64 && mimeType) { parts.push({ inlineData: { data: imageBase64, mimeType: mimeType } }); } // 呼叫 Google AI const result = await model.generateContent({ contents: [{ role: 'user', parts }] }); const responseText = result.response.text(); // 將結果傳回給前端 res.json({ result: responseText }); } catch (error) { console.error('Gemini API Error:', error); res.status(500).json({ error: 'AI 轉換失敗', details: error.message }); } }); app.listen(port, () => { console.log(`Backend is running on port ${port}`); });
